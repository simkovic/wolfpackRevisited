model = """
data {
    int<lower=0> N; // number of subjects
    int<lower=0> T; // number of trials
    real px[N,T,4]; // predictor on x axis
    real py[N,T,4]; // predictor on y axis
    real ox[N,T]; // outcome on x axis
    real oy[N,T]; // outcome on y axis
    int<lower=0,upper=1> sel[N,T]; // indicator var to include item
}
parameters {
    real a1[N];
    real a2[N];
    real a3[N];
    real a4[N];
    real<lower=0> sigma[N];
}
model {
    real mx;
    real my;
    for (n in 1:N){
        for (t in 1:T){
            if (sel[n,t]==1){
                mx<-a2[n]*px[n,t,2]+a3[n]*px[n,t,3]+a4[n]*px[n,t,4];
                my<-a1[n]*py[n,t,1]+a2[n]*py[n,t,2]+a3[n]*py[n,t,3]+a4[n]*py[n,t,4];
                ox[n,t] ~ normal(mx,sigma[n]);
                oy[n,t] ~ normal(my,sigma[n]);
            }
        }
    }
}
"""
