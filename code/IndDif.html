Differences due to individual factors will be also corrected. Any analysis based on estimates from hierarchical models thus will be very conservative.

def pearsonrci(r,n):
    z= np.arctanh(r)
    sse= ((n-3)/1.06)**-0.5
    er= sse*stats.norm.ppf(0.975)
    ci=np.array([-er,er])+z
    return np.tanh(ci)
