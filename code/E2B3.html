model = """
data {
    int<lower=0> N; // number of subjects
    int<lower=0> T; // number of trials
    real dr[N,2,T]; // discrepancy
}
parameters {
    real<lower=-10,upper=10> mu[N,2];
    real<lower=0,upper=10> sigma[N,2];
    real<lower=-10,upper=10> mmu[2];
    real<lower=0,upper=10> smu[2];
}
model {  
    for (n in 1:N){
        mu[n,1]~normal(mmu[1],smu[1]);
        mu[n,2]~normal(mmu[2],smu[2]);
        for (t in 1:T){
            dr[n,1,t] ~ normal(mu[n,1],sigma[n,1]);
            dr[n,2,t] ~ normal(mu[n,2],sigma[n,2]);

}
"""
