model = """
data {
    int<lower=0> N; // number of subjects
    int<lower=0> T; // number of trials
    real owt[N,T]; // proportion of time spent in wolfpack areas
    real pmd[N,T]; // magnitude of physical displacement
}
parameters {
    real<lower=0,upper=1> a0[N];
    real<lower=-0.5,upper=0.5>  a1[N];
    real<lower=0,upper=100> nu[N];
    
    real<lower=0,upper=1>  ma0;
    real<lower=-0.5,upper=0.5> ma1;
    real<lower=0> sa0;
    real<lower=0> sa1;
}
model {
    real mu;
    a0~normal(ma0,sa0);
    a1~normal(ma1,sa1);
    nu~pareto(0.1,1.5);
    for (n in 1:N){
        for (t in 1:T){
            mu <- a0[n]+a1[n]*pmd[n,t];
            owt[n,t]~beta(mu*nu[n],(1-mu)*nu[n]);
        }
    }    
}
"""
dat = {'N': len(vpn),'T':T,'pmd':D[:,1,:,10],'owt':D[:,1,:,0]}
fit = pystan.stan(model_code=model, data=dat,iter=5000, chains=4,warmup=1000,thin=5)

model = """
data {
    int<lower=0> N; // number of subjects
    int<lower=0> T; // number of trials
    real<lower=-1,upper=1> px[N,T,4]; // predictor on x axis
    real<lower=-1,upper=1> py[N,T,4]; // predictor on y axis
    real<lower=-20,upper=20> ox[N,T]; // outcome on x axis
    real<lower=-20,upper=20> oy[N,T]; // outcome on y axis
    int<lower=0,upper=1> sel[N,T]; // indicator var to include item
    real<lower=-0.2,upper=0.35> pmd[N,T]; // magnitude of physical displacement
}
parameters {
    real<lower=-2,upper=2> a1[N];
    real<lower=-2,upper=2> a2[N];
    real<lower=-2,upper=2> a3[N];
    real<lower=-2,upper=2> r0[N];
    real r1[N];
    real<lower=0,upper=4> sigma[N];
    real ma[5];
    real<lower=0,upper=4> sa[5];
}

    real mx;
    real my;
    a1~normal(ma[1],sa[1]);
    a2~normal(ma[2],sa[2]);
    a3~normal(ma[3],sa[3]);
    r0~normal(ma[4],sa[4]);
    r1~normal(ma[5],sa[5]);
